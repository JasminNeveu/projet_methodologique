\begin{frame}{Hypothesis testing}

\begin{block}{Type I error}
A valid test satisfies:
  \begin{equation*}
    \mathbb{P}(\text{Reject } \mathcal{H}_0) \leq \alpha.
  \end{equation*}
\end{block}
\pause
\vspace{-0.2cm}
\begin{block}{Testing via $p$-values}
The quantity $p$ is a $p$-value for $\mathcal{H}_0$ if:
\begin{equation*}
    \mathbb{P}_{\mathcal{H}_0}(p \leq u) \leq u, \quad \forall\, u \in [0,1].
    \qquad \text{(\textit{super-uniformity})}
  \end{equation*}
\end{block}
\pause
\vspace{-0.2cm}
\begin{block}{Rejection rule}
The rejection rule $\mathds{1}\{p \leq \alpha\}$ controls type I error at level $\alpha$.
\end{block}
\pause
    
\end{frame}

\begin{frame}{Interpreting a $p$-value ECDF}

  \begin{itemize}
    \item valid $p$-value : ECDF lies \textit{below} the diagonal
    \item Deviation \textit{above}: type I error inflation
  \end{itemize}

\begin{frame}{Selective inference}
\blue{Classical inference :}
  \begin{center}
    Fix $\mathcal{H}_0$ \textit{a priori} $\longrightarrow$ Test $\mathcal{H}_0$ on $\mathbf{X}$
  \end{center}
\pause
  \vspace{0.5em}

  \blue{Post-clustering inference :}
  \begin{center}
    Cluster $\mathbf{X}$ $\longrightarrow$ $\mathcal{H}_0$ defined from $\mathbf{X}$ $\longrightarrow$ Test $\mathcal{H}_0$ on $\mathbf{X}$
  \end{center}
\pause
  \vspace{0.5em}

  $\Rightarrow$ \textbf{Double-dipping :} $\mathbf{X}$ is used both to \textit{select} and to \textit{test} $\mathcal{H}_0$

  \vspace{0.5em}

  \blue{Analogy with supervised learning :} using the same data for training and evaluation leads to overfitting : the model captures noise rather than signal

  \vspace{0.4em}

  \begin{alertblock}{}
    \centering
    This is \textbf{selective inference :} a different framework is needed
  \end{alertblock}

\end{frame}

\begin{frame}{Addressing Selective Inference}

  \begin{block}{Simultaneous Inference}
\begin{itemize}
    \item Controls the family-wise error rate:
      \begin{equation*}\mathbb{P}(\text{at least 1 false positive amoung all hypothesis}) \leq \alpha\end{equation*}
    \item Too conservative, no post-clustering method proposed to date
\end{itemize}
\end{block}
\vspace{1em}
\begin{block}{Conditional Inference \hfill {\color{black}$\leftarrow$ \textit{this work}}}
  \begin{itemize}
    \item Controls the selective type I error:
      \begin{equation*}\mathbb{P}\!\left(\text{Reject } \mathcal{H}_0(X) ~\text{being true} \mid \mathcal{H}_0 \text{ selected}\right) \leq \alpha\end{equation*}
    \item State-of-the-art framework, basis of Gao et al.'s approach
  \end{itemize}
\end{block}

\end{frame}

\section{Gao \textit{et al.}'s approach}

\sectionslide{Gao \textit{et al.}'s approach}

\begin{frame}{Notations}
  \begin{itemize}
  \item Let $\mathbf{X} = (X_{ij})_{ij} \in \mathbb{R}^{n \times p}$ be the design matrix
    \vspace{0.9em}
  \item Let $C$ a clustering algorithm and $G \in C(\mathbf{X})$ a cluster obtained by the C algorithm on $\mathbf{X}$
    \vspace{0.9em}
  \item Let $\boldsymbol{\mu} = (\mu_{ij})_{ij}$ be the mean matrix of $\mathbf{X}$ i.e $\mu_{ij} = \mathbb{E}[X_{ij}]$
    \vspace{0.9em}
\item Let $\bar{\mu}_G = \dfrac{1}{|G|} \displaystyle\sum_{i \in G} \mu_i \in \mathbb{R}^p\quad\text{and}\quad\bar{X}_G = \dfrac{1}{|G|} \displaystyle\sum_{i \in G} X_i \in \mathbb{R}^p$, for $G$ a subset of $\{1,...,n\}$
  \vspace{0.5em}
  \begin{block}{Null hypothesis}
\begin{equation*}
    \mathcal{H}_0^{\{C_1,C_2\}} : \bar{\mu}_{C_1} = \bar{\mu}_{C_2} \iff \nu^\top\boldsymbol{\mu} = 0
  \end{equation*}
  \vspace{0.2em}
  where $\nu_i = \frac{1}{|C_1|}$ if $i \in C_1$, $-\frac{1}{|C_2|}$ if $i \in C_2$, $0$ otherwise
  \end{block}

\end{itemize}
\end{frame}

\begin{frame}{Gao \textit{et al.}'s approach}

  \blue{Model :}
  \begin{equation*}
    \mathbf{X} \sim \mathcal{MN}_{n\times p}(\boldsymbol{\mu}, \mathbb{I}_n, \sigma^2\mathbb{I}_p)
  \end{equation*}
  i.e. rows of $\mathbf{X}$ are independent $\mathcal{N}_p(\mu_i, \sigma^2\mathbb{I}_p)$ \textbf{independent features}

  \vspace{0.5em}

  \blue{Goal :} test equality of means between two clusters $C_1, C_2 \in \mathcal{C}(\mathbf{X})$
  \begin{equation*}
    \mathcal{H}_0^{\{C_1,C_2\}} : \bar{\mu}_{C_1} = \bar{\mu}_{C_2} \iff \nu^\top\boldsymbol{\mu} = 0
  \end{equation*}
  where $\nu_i = \frac{1}{|C_1|}$ if $i \in C_1$, $-\frac{1}{|C_2|}$ if $i \in C_2$, $0$ otherwise

\end{frame}




\begin{frame}{Selective type I error for clustering}

  \begin{block}{Selective type I error :}
  \begin{equation*}
    \mathbb{P}_{\mathcal{H}_0^{\{C_1,C_2\}}}\bigl(\text{Reject } \mathcal{H}_0^{\{C_1,C_2\}} \mid C_1,C_2 \in \mathcal{C}(\mathbf{X})\bigr) \leq \alpha
  \end{equation*}
\end{block}

  \begin{block}{Ideal $p$-value :}
  \begin{equation*}
    p_{ideal}(\mathbf{x}) = \mathbb{P}_{\mathcal{H}_0^{\{C_1,C_2\}}}\bigl(\|\nu^\top\mathbf{X}\|_2 \geq \|\nu^\top\mathbf{x}\|_2 \mid C_1,C_2 \in \mathcal{C}(\mathbf{X})\bigr)
  \end{equation*}
\end{block}
\begin{block}{Problem :}
    $p_{ideal}$ depends on unknown parameters, intractable in practice

  $\Rightarrow$ Add additional events $\xi(\mathbf{X},\mathbf{x})$ to the conditioning set.
\end{block}
\begin{block}{Tractable $p$-value :}
  \vspace{-0.8em}
  \begin{equation*}
    p(\mathbf{x}) = \mathbb{P}_{\mathcal{H}_0^{\{C_1,C_2\}}}\Bigl(\norm{\nu^T\mathbf{X}}_2 \geq \norm{\nu^T\mathbf{x}}_2 ~ | ~ C_1,C_2 \in \mathcal{C}(\mathbf{X}), \xi(\mathbf{X},\mathbf{x})\Bigr)
  \end{equation*}
\end{block}
\end{frame}


\begin{frame}{Additional events}

  \begin{block}{Decomposition of $\mathbf{X}$}
    \begin{equation*}
      \mathbf{X} = \projnu \mathbf{X} + \frac{\nu}{\|\nu\|_2^2} \operatorname{dir}(\nu^\top \mathbf{X}) \|\nu^\top \mathbf{X}\|_2
    \end{equation*}
  \end{block}

  \begin{equation*}
    \text{with} ~\projnu = \mathbb{I}_n - \frac{\nu\nu^\top}{\|\nu\|_2^2} \qquad \text{and} \qquad \operatorname{dir}(z) = \frac{z}{\|z\|_2}, \quad z \in \mathbb{R}^p
  \end{equation*}
\pause
  \vspace{1em}

  \blue{Additional events:}
  \begin{equation*}
    \xi(\mathbf{X},\mathbf{x}) = \bigl\{ \projnu \mathbf{X} = \projnu \mathbf{x}, \quad \operatorname{dir}(\nu^\top \mathbf{X}) = \operatorname{dir}(\nu^\top \mathbf{x})\bigr\}
  \end{equation*}

\end{frame}


\begin{frame}{Tractable $p$-value}

  \begin{block}{Proposition} Under $\mathcal{H}_0$:
  \begin{center}
    (i) $\|\nu^\top \mathbf{X}\|_2 \indep \pi_\nu^\perp \mathbf{X}$ \qquad 
    (ii) $\|\nu^\top \mathbf{X}\|_2 \indep \operatorname{dir}(\nu^\top \mathbf{X})$
  \end{center}

\end{block}
  \vspace{0.5em}

  Following from the previous result, we can rewrite the $p$-value as:
  
  \begin{multline*}
    p(\mathbf{x}) = \mathbb{P}_{\mathcal{H}_0} \Bigl( \|\nu^\top \mathbf{X}\|_2 \geq \|\nu^\top \mathbf{x}\|_2 ~\Big|~ \\
    C_1, C_2 \in \mathcal{C} \Bigl( \pi_\nu^\perp \mathbf{x} + \frac{\nu}{\|\nu\|^2_2} \operatorname{dir}(\nu^\top \mathbf{x}) \|\nu^\top \mathbf{X}\|_2 \Bigr) \Bigr).
  \end{multline*}

\end{frame}


\begin{frame}{Truncation set $\mathcal{S}$}

  Let $\phi = \|\nu^\top \mathbf{X}\|_2$ 
 \begin{block}{Distribution of the test statistic} Under $\mathcal{H}_0^{\{C_1,C_2\}}$:
  \begin{equation*}
    \phi \sim \|\nu\|_2\,\sigma\,\chi_p
  \end{equation*}
\end{block}

\vspace{0.4em}
Let $\mathbf{x}'(\phi) = \pi_\nu^\perp \mathbf{x} + \dfrac{\nu}{\|\nu\|_2^2} \operatorname{dir}(\nu^\top \mathbf{x})\phi$
  \vspace{0.4em}

  \begin{equation*}
    p(\mathbf{x}) = \mathbb{P}_{\mathcal{H}_0}\Bigl(\phi \geq \|\nu^\top \mathbf{x}\|_2 ~\Big|~ \phi \in \mathcal{S}\bigl(\mathbf{x};\{C_1,C_2\}\bigr)\Bigr)
  \end{equation*}

  \vspace{0.3em}

  where
  \begin{equation*}
    \mathcal{S}\bigl(\mathbf{x},\{C_1,C_2\}\bigr) = \Bigl\{\phi \geq 0 : C_1,C_2 \in \mathcal{C}\bigl(\mathbf{x}'(\phi)\bigr)\Bigr\}
  \end{equation*}

  \begin{block}{Interpretation}
    $\mathcal{S}$ is the set of perturbations of $\mathbf{x}$ that preserve the clusters $C_1, C_2$
  \end{block}
   



\end{frame}
\begin{frame}{Interpretation of $\mathbf{x}'(\phi)$}

  \begin{equation*}
    \text{Let} ~\mathbf{x}'(\phi) = \mathbf{x} + \frac{\nu}{\|\nu\|_2^2}\operatorname{dir}(\nu^\top \mathbf{x})\,(\phi - \|\nu^\top \mathbf{x}\|_2)
  \end{equation*}
  \vspace{1em}
  $\mathbf{x}'(\phi)$ is a perturbation of $\mathbf{x}$ along the direction $\nu$ -- clusters are pulled together or pushed apart depending on $\phi$

  \vspace{1em}

\begin{frame}{Theorem 1 — Gao \textit{et al.}}

  For any $C_1, C_2 \subseteq \{1,\ldots,n\}$ and any realization $\mathbf{x}$ of $\mathbf{X}$:
  \begin{equation*}
    p(\mathbf{x}) = 1 - \mathbb{F}\bigl(\|\nu^\top \mathbf{x}\|_2;\,\|\nu\|_2 \sigma;\,\mathcal{S}\bigl(\mathbf{x};\{C_1,C_2\}\bigr)\bigr)
  \end{equation*}

  where $\mathbb{F}(t; c; \mathcal{S})$ is the CDF of a $c\cdot\chi_p$ random variable truncated to the set $\mathcal{S}$

  \vspace{0.7em}

  Furthermore, if $\mathcal{H}_0^{\{C_1,C_2\}}$ is true:
  \begin{equation*}
    \mathbb{P}_{\mathcal{H}_0^{\{C_1,C_2\}}}\Bigl(p(\mathbf{x}) \leq \alpha ~\Big|~ C_1,C_2 \in \mathcal{C}(\mathbf{X})\Bigr) = \alpha, \quad \forall\, \alpha \in [0,1]
  \end{equation*}
\end{frame}

\begin{frame}{Suitability of Gao \textit{et al.}'s approach}

  \blue{In practice:} keep as many PCs as possible to minimize information loss

\pause

  \vspace{0.4em}
  \blue{Power of a test:} probability of rejecting $\mathcal{H}_0$ when it is false\\
  $\Rightarrow$ a useful test must be both \textit{valid} and \textit{powerful}

  \vspace{0.4em}
\pause
  \begin{alertblock}{}
    \centering

\section{Discussion}

\begin{frame}{Discussion}
  \begin{itemize}
    \item Gao \textit{et al.}'s approach: seminal contribution to the conditional inference framework
      \pause
    \item First procedure to test the equality of means between clusters estimated via HAC, under a Gaussian model with independent features
      \pause
    \item Asymptotically valid when $\sigma$ is overestimated
      \pause
    \item When the independence assumption is violated, the $p$-value deviates from super-uniformity and the deviation increases with both $\rho$ and $p$
      \pause
    \item These limitations persist on real data, yet the method successfully recovers broad population structure
      \pause
    \item Gonz\'{a}lez-Delgado \textit{et al.} proposed an extension that remains valid under arbitrary dependence structures between observations and features
  \end{itemize}
\end{frame}
